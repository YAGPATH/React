import React from 'react'

function header() {
  return (
    <header style ={{
      backgroundColor: 'grey',
      color: 'white'
    }}>
        <h1>Groceries List</h1>
    </header>
  )
}

export default header




- we can see that we have styled the header with outline styling. The type of styling is same for every component.
